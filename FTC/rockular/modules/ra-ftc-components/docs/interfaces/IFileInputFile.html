<ra-ftc-file-input> provides a file input component.

Configuration

File Input component takes 12 parameters

<ra-ftc-file-input
    [appearance]="appearance" <!-- string, either 'chips' or 'input' -->
    [label]="label" <!-- string, label to describe the input field's purpose -->
    [color]="color" <!-- ThemePalette, either 'accent', 'warn' or 'primary -->
    [multiple]="allowMultiple" <!-- boolean, true allows the input of more than just 1 file -->
    [files]="passedFiles" <!-- IFileInputFile[], array of files passed to the component  -->
    [allowedMimeTypes]="allowedMimeTypes" <!-- string, supported MIME types. Different types have to be comma separated. Unlike RFC2046 it supports wildcards such as image/* -->
    [allowedMimeTypesHint]="allowedMimeTypesHint" <!-- string, hint to allowed MIME types -->
    [disabled]="disabled" <!-- boolean, if true disables the input field -->
    (filesAdded)="onFilesAdded($event)" <!-- event listener, fires when a file is added   -->
    (filesRemoved)="onFilesRemoved($event)" <!-- event listener, fires when a file is removed -->
    (filesChanged)="onFilesChanged($event)" <!-- event listener, fires when a file is changed (only works when multiple === true) -->
    (disallowedFilesAdded)="onDisallowedFilesAdded($event)"> <!-- event listener, fires when a user attempts to add a disallowed file type -->
</ra-ftc-file-input>

Interfaces

interface IFileInputFile {
    name: string;
    size?: number;
    lastModified?: number; // this is not available on IE 11
    lastModifiedDate?: Date;
    type?: string;
}

Examples

appearance = 'chips';
label = 'Label';
color: ThemePalette = 'accent';
disabled = false;
allowMultiple = false;
allowedMimeTypes = 'image/*';
allowedMimeTypesHint = 'Only image files are supported';

    console.log(f);

    console.log(f);
}

}
