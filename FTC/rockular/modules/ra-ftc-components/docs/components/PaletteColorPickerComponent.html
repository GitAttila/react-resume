PaletteColorPickerComponent provides visual representation for palette color picker.

Palette color picker

This module provides methods and definitions to show selected color and also allows user to pick a color from component configurable palettes. It is meant to be used with palette or palettes containing different colors, even between palettes.

Configuration

Input bindings

It is possible to set these configuration properties (as attribute property binding):

  • appearance - optional settings to allow set visualization as circle or square.
    const appearance: ColorPickerAppearance = 'square';  // default is 'circle'
  • palettes - array of palettes with palette label and colors.
    const palettes: IPaletteColorPickerPalette[] = [
  {
      name: 'palette1',
      colors: ['#333', '#aaa']
  },
  {
      label: 'Palette 2',
      name: 'palette2',
      colorsPerRow: 3,
      colors: [
          '#1F497D',
          '#4F81BD',
          '#C0504D',
          '#9BBB59',
          '#8064A2',
          '#4BACC6'
      ]
  },
  {
      label: 'Palette 3',
      name: 'palette3',
      colors: [
          { h: 10, s: 50, l: 90, a: 0.8 },
          { r: 80, g: 190, b: 255, a: 0.5 },
          '#EEECE1',
          '#1F497D',
          '#4F81BD',
          '#C0504D',
          '#9BBB59',
          '#8064A2',
          '#4BACC6',
          '#F79646',
          '#FFFFFF',
          '#000000',
          '#E3DED1',
          '#323232',
          '#F07F09',
          '#9F2936',
          '#1B587C',
          '#4E8542',
          '#604878',
          '#C19859'
      ]
  }
];
  • value - actual color value
    // value has no default value
// value of type <ColorPickerInput> allows: string | IColorPickerRGB | IColorPickerRGBA | IColorPickerHSLA IColorPickerHSLA | TinyColor
const colorValue: string = '#C19859'; // also 'rgb(10, 25, 30)', 'rgba(10, 25, 30, 1)' is supported
const colorValue: IColorPickerHSL = {h: 10, s: 50, l: 90};
const colorValue: IColorPickerHSLA = {h: 10, s: 50, l: 90, a: 0.8};
const colorValue: IColorPickerRGB = {r: 80, g: 190, b: 255};
const colorValue: IColorPickerRGBA = {r: 80, g: 190, b: 255, a: 0.5};
const colorValue: IColorPickerHSV = {h: 10, s: 50, v: 90};
const colorValue: IColorPickerHSVA = {h: 10, s: 50, v: 90, a: 0.5};
const colorValue: TinyColor = new TinyColor('rgb(10, 25, 30)');
  • valuePaletteName - palette from which to select passed value, one of passed palettes 'name'
    const colorPaletteName: string = 'palette1';
  • disabled - flag indicating disabled state of component, user is not able to select color
    const disabled: boolean = false; // default as false
  • showDropdown - flag indicating to show dropdown
    const showDropdown: boolean = false; // default as false
  • required - flag indicating that user is not able to unselect previously selected color
    const required: boolean = false; // default as false

Output bindings

  • colorSelected - triggered when color is selected from one of palettes, this is of type PaletteColor
    onColorSelected(output) {
  // output ~
  output = {
      hex: '#c0504d', 
      rgb: { r: 192, g: 80, b: 77, a: 1 }, 
      rgbString: 'rgb(192, 80, 77)', 
      hsl: { h: 1.5652173913043497, s: 0.47717842323651455, l: 0.5274509803921569, a: 1 }, 
      hslString: 'hsl(2, 48%, 53%)', 
      hsv: { h: 1.5652173913043497, s: 0.5989583333333334, v: 0.7529411764705882, a: 1 }, 
      hsvString: 'hsl(2, 48%, 53%)' 
  };
}
  • opened - triggered when popup is opened
  • closed - triggered when popup is closed
