StringValuePickerComponent component which content in ng-template is after click on input or button displayed in dialog. Interaction of input and dialog content has to be specified by developer. Look at examples.

String value picker

This module provides methods and definition to show content inside dialog and allows to set value of the component input.

Configuration

Input bindings

It is possible to set these configuration properties (as attribute property binding):

  • value - string value displayed inside string value picker input.
    const value: string = '';
  • placeholder - input placeholder text. Use translation service to localize placeholder text.
    const placeholder: string = ''; // default as RA_FTC_COMPONENTS.RA_FTC_STRING_VALUE_PICKER.DEFAULT_PLACEHOLDER locale
  • disabled - flag indicating disabled state of component
    const disabled: boolean = false; // default as false

Output bindings

  • opened - triggered when dialog is opened
  • closed - triggered when dialog is closed

Examples

<ra-ftc-string-value-picker [value]="pickerValue" [disabled]="disabled" placeholder="">
              <ng-template>
                <div style="width: 30vw;">
                  <div style="font-weight: 700;" fxLayout="row" fxLayoutAlign="center center">
                    <h2>Pick value</h2>
                  </div>
                  <mat-radio-group fxLayout="column" [value]="pickerValue">
                    <mat-radio-button style="margin: 5px;" *ngFor="let shoe of typesOfShoes" [value]="shoe">
                      {{shoe}}
                    </mat-radio-button>
                  </mat-radio-group>
                  <div fxLayout="row" fxLayoutAlign="end center">
                    <button mat-button (click)="setPickerValue(_radiogroup.value)">Ok</button>
                    <button mat-button (click)="closeDialog()">Cancel</button>
                  </div>
                </div>
              </ng-template>
</ra-ftc-string-value-picker>
@Component({
    selector: 'app-string-value-picker-test',
    templateUrl: './string-value-picker-test.component.html'
})
export class StringValuePickerTestComponent {
    @ViewChild(StringValuePickerComponent) _strValuePicker: StringValuePickerComponent;
    @ViewChild(MatRadioGroup) _radiogroup: MatRadioGroup;

    pickerValue: string;

    typesOfShoes: string[] = ['Boots', 'Clogs', 'Loafers', 'Moccasins', 'Sneakers'];

    closeDialog() {
      this._strValuePicker.closeDialog();
    }

    setPickerValue(value: string) {
      this.pickerValue = value;
      this.closeDialog();
    }
}
