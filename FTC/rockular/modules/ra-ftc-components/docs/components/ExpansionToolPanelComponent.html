<ra-ftc-expansion-tool-panel> provides an expandable panel.

Expansion-tool-panel content

Each expansion-tool-panel is an expandable panel which can include any content. Supports positionning at left or right side.

Configuration

Component can be configured by passing variable in config attribute of type IExpansionToolPanelConfig.

<ra-ftc-expansion-tool-panel [config]="config"></ra-ftc-expansion-tool-panel>
const config: IExpansionToolPanelConfig = {
        defaultState: 'collapsed', // expanded or collapsed, default as collapsed
        positionBehavior: 'left', // left or right, sets behavior of component in layout
        collapsedWidth: '72px', // width to which component collapses to, default as 72px
        expandedWidth: '250px', // width to which component expands to, default as 250px
        toolbar: {
            visible: true, // true or false, if toolbar has to be visible, it contains expand/collapse button
            position: 'top' // 'bottom' or 'top', where toolbar will appear if visible
        }
    };
