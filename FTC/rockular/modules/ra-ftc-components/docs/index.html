@ra-wcftcloud/ra-ftc-components

About

This javascript library consists of common components to be used across any client application.

Adding to an Angular CLI app

When adding this library to new application please check out the Angular CLI.

Use npm to install the @ra-wcftcloud/ra-ftc-components library

NOTE: The library should be installed using over SSH connection using SSH key from your GitLab profile. How to generate SSH key.

Install with npm using the release version (1.0.0 in the examples below).

npm install git+ssh://git@gitlabpra.ra.rockwell.com:CAT_WCFTCloud/releases/ra-ftc-components.git#1.0.0

or add the @ra-wcftcloud/ra-ftc-components library to your package.json

"@ra-wcftcloud/ra-ftc-components": "git+ssh://git@gitlabpra.ra.rockwell.com:CAT_WCFTCloud/releases/ra-ftc-components.git#1.0.0"

Add peer dependencies

Add peer dependencies to your package.json file:

"@angular/common": "^6.0.1",
"@angular/core": "^6.0.1",
"@angular/flex-layout": "^6.0.0-beta.16",
"@angular/material": "^6.0.1",
"@asymmetrik/ngx-leaflet": "^3.0.2",
"@asymmetrik/ngx-leaflet-markercluster": "^1.0.0",
"@ctrl/tinycolor": "^2.2.0",
"@daimoonis/ngx-color": "^0.0.8",
"@ngx-translate/core": "^10.0.2",
"@ra-web-tech-ui-toolkit/common-utils": "^1.0.0",
"lodash": "^4.17.10",
"leaflet": "^1.3.1",
"leaflet.markercluster": "^1.3.0",
"material-colors": "^1.2.6",

Run npm install to install all of the packages that you just added.

Add library to your app's module file

In your app, import any library module, e.g.:

import { RaFtcStringValuePickerModule } from '@ra-wcftcloud/ra-ftc-components';

@NgModule({
  ...
  imports: [
    ...
    RaFtcStringValuePickerModule
  ],
  ...
}

Include library themes

For correct visual you should apply styles in your app style.scss file, import the styles from library sources:

@import "~@ra-wcftcloud/ra-ftc-components/styles";

@include ra-ftc-components-theme($theme);

Example using WTUI predefined theme:

@import "~@ra-web-tech-ui-toolkit/common-utils/styles";
// source with $theme, $custom-typography definitions and more:
@import "~@ra-web-tech-ui-toolkit/common-utils/themes/variables/rockwell-blue";
@import "~@ra-wcftcloud/ra-ftc-components/styles";

$app-typography-config: mat-typography-config();

@include mat-core($app-typography-config);
@include common-utils-theme($theme, $app-typography-config);
@include ra-ftc-components-theme($theme);

Example using pure Angular material theme:

// in material theming there are definitions for palettes, typography and more:
@import "~@angular/material/theming";
@import "~@ra-wcftcloud/ra-ftc-components/styles";

$app-typography-config: mat-typography-config($input: mat-typography-level(16px, 1.3, 400), $body-1: mat-typography-level(16px, 21px, 400));

@include mat-core($app-typography-config);

// example of a class to be applied to body or application's root element, 
// also in such case this has to be applied to angular material overlays
.indigo-pink-theme {
    $primary: mat-palette($mat-indigo);
    $accent: mat-palette($mat-pink, A200, A100, A40);
    $theme: mat-light-theme($primary, $accent);
    @include angular-material-theme($theme);
    @include common-utils-theme($theme, $app-typography-config);
    @include ra-ftc-components-theme($theme);
    // example of custom styles:
    &.primary-color {
        color: mat-color($primary, 500);
    }
    &.primary-background-color {
        background-color: mat-color($primary, 500);
    }
}
