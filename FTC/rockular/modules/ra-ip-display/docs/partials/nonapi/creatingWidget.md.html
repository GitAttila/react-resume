Creating widget

When implementing a widget, widget should also define its own constant service with specific pattern ConstSvc and implement these methods: getSchema (returns widget schema), getExamples (returns widgets examples). In case there are constants needed, there may also be defined getConstant method.

Example:

```
    angular.module('myModule')
        .service("myWidgetConstSvc", [function () {
                "use strict";
                var constants = {};
                constants.myConstant = 'example';

                /**
                 * @description Function to return predefined schema only. Schema can have translations included, which
                 * cannot be used in service initialization state.
                 * @returns {Object} - schema
                 */
                this.getSchema = function getSchema() {
                    return {
                        "name": "My Widget",
                        "supportMultiple": false,
                        "schemaVersion": "2.0.0",
                        "allowPreview": false,
                        "schema": []
                    };
                };

                /**
                 * @description Function to return predefined examples only.
                 * @returns {Array}
                 */
                this.getExamples = function getExamples() {
                    return [];
                };

                /**
                 * @description Function to return predefined static constant.
                 * @param {string} name - name of constant
                 * @returns {any} - predefined constant
                 */
                this.getConstant = function getConstant(name) {
                    return angular.copy(constants[name]);
                };
            }
        ])
        .directive('myWidget', ['myWidgetConstSvc', function (myWidgetConstSvc) { 
            var myConstant = myWidgetConstSvc.getConstant('myConstant')
            ...
        });
```
